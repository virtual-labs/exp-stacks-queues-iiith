1 Introduction

This document contains the JavaScript code (Canvas API) for the requirement artifact - queues using arrays. An array is displayed into which a user can enqueue values or from which a user can dequeue values. A head variable keeps track of the index of the current head of the queue and the tail variable keeps track of the index of the tail of the queue.

2 queue_Array class

  • Creates queue_Array class, which has all the variables used for canvas attributes and its corresponding

positions on the canvas .

  • We use "let" to create an object of this class
  • Here we initialize the canvas and define a class name queue_Array
  • Font of numbers to be inserted in the queue is defined by txtSize
  • Number of array blocks is defined by rectNo
  • Dimensions of array rectangle is defined by rectHeight and rectWidth
  • Starting position of array coordinates are given by rectStartx and rectStarty
  • Position of the indices of the array are given by indexPosx and indexPosy
  • Head of queue (initialized to -1) is defined by head and its coordinates are given by headPosx and headPosy
  • Tail of queue (initialized to 0) is defined by tail and its coordinates are given by tailPosy
  • Position of arrow pointing to head are given by hArrowStartx , hArrowStarty , hArrowEndx and hArrowEndy
  • Position of arrow heads to the head of queue are given by hArrowHeadOffsetx and hArrowHeadOffsety
  • Coordinates of 'Head' text below arrow are given by hArrowHeadx and hArrowHeady
  • Position of arrow pointing to tail are given by tArrowStartx , tArrowStarty , tArrowEndx and tArrowEndy
  • Position of arrow heads to the tail of queue are given by tArrowHeadOffsetx and tArrowHeadOffsety
  • Coordinates of 'Tail' text below arrow are given by tArrowTailx and tArrowTaily
  • Values of Queue are stored in values array
  • queuearray_artefact is the object of this class
 var canvas = document.getElementById("queueArray");
canvas.width = 1500;
canvas.height = 340;
var ctx = canvas.getContext("2d");
class queue_Array{
    constructor(){
        this.lineDist = 3;
        this.txtSize = 25;
        this.rectNo = 8;
        this.lineNo = this.rectNo - 1;
        this.rectWidth = 95;
        this.rectHeight = 80;
        this.rectStartx = 250;
        this.rectStarty = 120;
        this.indexPosx = this.rectStartx;
        this.indexPosy = this.rectStarty + this.rectHeight + 40;
        this.head = 0; 
        this.headPosy = this.indexPosy + 250;
        this.tail = 0; 
        this.tailPosy = this.headPosy + 50;
        this.hArrowStartx = this.rectStartx + this.rectWidth/2;
        this.hArrowStarty = this.rectStarty - 30;
        this.hArrowEndx = this.hArrowStartx;
        this.hArrowEndy = this.hArrowStarty - 50;
        this.hArrowHeadOffsetx = 10;
        this.hArrowHeadOffsety = 10;
        this.hArrowHeadx = 10;
        this.hArrowHeady = 10;
        this.tArrowStartx = this.rectStartx + this.rectWidth/2;
        this.tArrowStarty = this.indexPosy + 10;
        this.tArrowEndx = this.tArrowStartx;
        this.tArrowEndy = this.tArrowStarty + 50;
        this.tArrowHeadOffsetx = 10;
        this.tArrowHeadOffsety = 10;
        this.tArrowTailx = 10;
        this.tArrowTaily = 20;
        this.errorPosy = this.rectStarty + this.rectHeight/2;
        this.values = [];
    };
};

5 Draw the array structure

Function to draw the basic array structure and base. 

function drawQueueStructure() {
ctx.beginPath();
ctx.rect(queuearray_artefact.rectStartx, queuearray_artefact.rectStarty, queuearray_artefact.rectWidth*queuearray_artefact.rectNo + queuearray_artefact.lineNo*queuearray_artefact.lineDist, queuearray_artefact.rectHeight);
ctx.strokeStyle = "#288ec8";
ctx.stroke();
                
for(var i=1; i<queuearray_artefact.rectNo; i++) 
{
    ctx.rect(queuearray_artefact.rectStartx+(i*queuearray_artefact.rectWidth)+((i-1)*queuearray_artefact.lineDist), queuearray_artefact.rectStarty, queuearray_artefact.lineDist, queuearray_artefact.rectHeight);
    ctx.strokeStyle = "#288ec8";
    ctx.stroke();
}
               
ctx.closePath();

8 Shade head and tail of queue

Color shade the head and tail of queue in array. 

function writeNumbers() {
clearCanvas();
drawQueueStructure();

if(queuearray_artefact.head > -1)
{
    ctx.beginPath();
    ctx.fillStyle = "#a4c652";

Function to write all the values present inside the queue into the array displayed to user. 

ctx.beginPath();    

if(queuearray_artefact.head > -1 && queuearray_artefact.values.length > 0){

    if(queuearray_artefact.tail > queuearray_artefact.head) {

        ctx.beginPath();

        {

            txtWidth = ctx.measureText(txt).width;
                
            txtX = queuearray_artefact.rectStartx + (i*queuearray_artefact.rectWidth) + (queuearray_artefact.rectWidth-txtWidth)/2 + (i*queuearray_artefact.lineDist);
            txtY = queuearray_artefact.rectStarty + queuearray_artefact.rectHeight - ((queuearray_artefact.rectHeight-queuearray_artefact.txtSize)/2);
                            
            ctx.fillText(txt, txtX, txtY);  
        }
    }
    else {

        ctx.beginPath();
        for(var i=queuearray_artefact.head+1; i<queuearray_artefact.rectNo; i++)
        {
            ctx.fillStyle = "#288ec8";
            ctx.fillRect(queuearray_artefact.rectStartx+(i*queuearray_artefact.rectWidth)+(i*queuearray_artefact.lineDist), queuearray_artefact.rectStarty, queuearray_artefact.rectWidth, queuearray_artefact.rectHeight);
        }
        for(var i=0; i<queuearray_artefact.tail; i++)
        {
            ctx.fillStyle = "#288ec8";
            ctx.fillRect(queuearray_artefact.rectStartx+(i*queuearray_artefact.rectWidth)+(i*queuearray_artefact.lineDist), queuearray_artefact.rectStarty, queuearray_artefact.rectWidth, queuearray_artefact.rectHeight);
        }
        ctx.closePath();

        var count = 0;

        for(var i=queuearray_artefact.head; i<queuearray_artefact.rectNo; i++) 
        {
            ctx.font = "25px Arial"
            ctx.fillStyle = "white";
                
            txt = queuearray_artefact.values[count].toString();
            txtWidth = ctx.measureText(txt).width;

            txtX = queuearray_artefact.rectStartx + (i*queuearray_artefact.rectWidth) + (queuearray_artefact.rectWidth-txtWidth)/2 + (i*queuearray_artefact.lineDist);
            txtY = queuearray_artefact.rectStarty + queuearray_artefact.rectHeight - ((queuearray_artefact.rectHeight-queuearray_artefact.txtSize)/2);
                            
            ctx.fillText(txt, txtX, txtY);

        }

        for(var i=0; i<queuearray_artefact.tail; i++)

13 Function enqueue

  • Function to enqueue the entered element and display resulting array on screen.
function enqueue(){
var value = document.getElementById('numbers-qa').value;

    if((queuearray_artefact.tail + 1) % queuearray_artefact.rectNo === queuearray_artefact.head) document.getElementById("insqc").innerHTML = "Queue is full";
    else if (value==="" || value===null) 
    {
        document.getElementById("insqc").innerHTML = "Please enter the element which you want to add to the queue";
        return;
    }
    else
    {
        queuearray_artefact.values.push(value);
        if(queuearray_artefact.head===-1) queuearray_artefact.head++;
        queuearray_artefact.tail = (queuearray_artefact.tail + 1) % queuearray_artefact.rectNo;
        document.getElementById("insqc").innerHTML = value + " is enqueued";

        writeNumbers();
    }
    document.getElementById('numbers-qa').value = "";
}

14 Function dequeue

Function to dequeue values from the queue and display resulting array on screen. 

function dequeue(){
if((queuearray_artefact.head === -1 && queuearray_artefact.tail === 0 )|| queuearray_artefact.head === queuearray_artefact.tail) document.getElementById("insqc").innerHTML = "Queue has no elements in it";
    else
    {
        temp = queuearray_artefact.values.shift();
        queuearray_artefact.head = (queuearray_artefact.head + 1) % queuearray_artefact.rectNo;
        document.getElementById("insqc").innerHTML = temp + " is dequeued";
        writeNumbers();
    }
}

15 Clearing the queue

Function to clear the queue array - remove all elements from the queue and reset head and tail variables. 

function clear(){
    clearCanvas();
    queuearray_artefact.values = [];
    queuearray_artefact.head = 0;
    queuearray_artefact.tail = 0;
    drawQueueStructure();
    writeNumbers();
    document.getElementById("insqc").innerHTML = "Queue is reset";
}
