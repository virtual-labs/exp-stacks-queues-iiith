1 Introduction

This document contains the JavaScript code (Canvas API) for the requirement artifact - stacks using arrays. An array is displayed into which a user can push values or from which a user can pop values. A top variable keeps track of the index of the current top of the stack.

2 stack_array class

  • Creates stack_array class, which has all the variables used for canvas attributes and its corresponding

positions on the canvas .

  • We use "let" to create an object of this class
  • Here we initialize the canvas and define a class name queue_Array
  • Font of numbers to be inserted is defined by txtSize
  • Distance between lines is defined by lineDist
  • Number of array blocks is defined by rectNo
  • Number of lines to seperate the array blocks is defined by lineNo
  • Dimensions of stack rectangle is defined by rectHeight and rectWidth
  • Starting position of stack coordinates is defined by rectStartx and rectStarty
  • Position of the indices of the array is defined by indexPosx and indexPosy
  • topOfStack Points to top of stack, initialized to -1 and its coordinate is defined by topPosy
  • Position of arrow pointing to top of stack is define by arrowStartx , arrowStarty , arrowEndx and arrowEndy
  • Position of arrow heads to the top of the stack is defined by arrowHeadOffsetx and arrowHeadOffsety
  • arrowTopx and arrowTopy are used to define the Coordinates of 'Top' text below arrow
  • Values of Stack are stored in values array
var canvas = document.getElementById("stackArray");
var ctx = canvas.getContext("2d");
canvas.width = 1500;
canvas.height = 280;
class stack_array {
    constructor() {
        this.txtSize = 25;
        this.lineDist = 3;
        this.rectNo = 8;
        this.lineNo = this.rectNo - 1;
        this.rectWidth = 95;
        this.rectHeight = 80;
        this.rectStartx = 360;
        this.rectStarty = 50;
        this.indexPosx = this.rectStartx;
        this.indexPosy = this.rectStarty + this.rectHeight + 40;
        this.topOfStack = -1;
        this.topPosy = this.indexPosy + 200;
        this.arrowStartx = this.rectStartx + (this.rectWidth / 2);
        this.arrowStarty = this.indexPosy + 20;
        this.arrowEndx = this.arrowStartx;
        this.arrowEndy = this.arrowStarty + 40;
        this.arrowHeadOffsetx = 7;
        this.arrowHeadOffsety = 10;
        this.arrowTopx = 5;
        this.arrowTopy = 25;
        this.values = [];
    };
};

8 Shade top of stack

function writeNumbers() {
clearCanvas();
drawArray();
                
if(stackarray_artefact.topOfStack > -1)

11 Function onpush

  • Function which gets called on clicking push button
  • Function to push values into the stack and display on screen(on clicking Push button).
function onpush(){
   var value = document.getElementById('numbers-stackarray').value;

    if(stackarray_artefact.values.length >= stackarray_artefact.rectNo) {
        document.getElementById("ins").innerHTML = "Stack overflow";
    }
    else if (value === "" || value === null) 
    {
        document.getElementById("ins").innerHTML = "Please enter an element in the textbox that you want to <b>PUSH</b> into the stack";
    }
    else
    {
            stackarray_artefact.values.push(value);
        stackarray_artefact.topOfStack++;
        writeNumbers();
        document.getElementById("ins").innerHTML = value + " is pushed into the stack." + "<br>The index of the top of stack:" + stackarray_artefact.topOfStack;            
    }
    document.getElementById('numbers-stackarray').value = "";
        
}

12 Function onpop

  • Function which gets called on clicking pop button
  • Function to pop values from the stack and display resulting array on screen(on clicking Pop button).
function onpop(){
   if(stackarray_artefact.topOfStack === -1){
        document.getElementById("ins").innerHTML = "Stack underflow";
    }
    else
    {
        temp = stackarray_artefact.values.pop();
            stackarray_artefact.topOfStack--;
                    
        writeNumbers();
        document.getElementById("ins").innerHTML = temp + " is popped from the stack." + "<br>The index of the top of stack:" + stackarray_artefact.topOfStack;     
    }
}

13 Function onclear

  • Function which gets called on clicking clear button
  • Function to clear the stack array - remove all elements from the stack and reset top variable(on clicking Clear button).
function onclear(){
    clearCanvas();
    stackarray_artefact.values = [];
    stackarray_artefact.topOfStack = -1;
    document.getElementById("ins").innerHTML = "Stack is cleared <br>  Index of the top of stack : -1";
    drawArray();
}

14 Adds all event handlers

  • A function that adds all the event handlers to the html document
function handlers()
{ 
 document.getElementById("push-button").onclick = function() { onpush(); };
 document.getElementById("pop-button").onclick = function() { onpop(); };
 document.getElementById("clear-button").onclick = function() { onclear(); };
};
